224 std::size_t computeHash(const std::string &filename, struct stat fileStat) {
225 // Get file size and modification time
226 size_t fileSize = fileStat.st_size;
227 time_t fileTime = fileStat.st_mtime;
228
229 // Read first 8 bytes of file
230 std::ifstream file(filename, std::ios::binary);
231 std::vector<char> buffer(8);
232 file.read(buffer.data(), 8);
233
234 // Hash file name, size, time, and header info
235 std::string fileHeader(buffer.begin(), buffer.end());
236 std::stringstream ss;
237 ss << fileSize << fileTime << fileHeader;
238 std::string fileString = filename + ss.str();
239 std::size_t fileHash = std::hash<std::string>{}(fileString);
240
241 return fileHash;
242 }
