152 HashResult computeHash(const std::string &filename) {
153
154 struct stat fileStat;
155
156 if (stat(filename.c_str(), &fileStat) != 0 &&
157 !std::filesystem::is_regular_file(filename)) {
158 auto error = "Error: " + filename + " is not a valid file.";
159 return {0, error};
160 }
161
162 // Get file size and modification time
163 size_t fileSize = fileStat.st_size;
164 time_t fileTime = fileStat.st_mtime;
165
166 // Read first 8 bytes of file
167 std::ifstream file(filename, std::ios::binary);
168 std::vector<char> buffer(8);
169 file.read(buffer.data(), 8);
170
171 // Hash file name, size, time, and header info
172 std::string fileHeader(buffer.begin(), buffer.end());
173 std::stringstream ss;
174 ss << fileSize << fileTime << fileHeader;
175 std::string fileString = filename + ss.str();
176 std::size_t fileHash = std::hash<std::string>{}(fileString);
177
178 return {fileHash, ""};
179 }
